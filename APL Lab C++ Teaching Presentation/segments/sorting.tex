\begin{frame}
    \frametitle{Iterators}

    
    \begin{itemize}
        \item Similar to pointers in C, C++ has Iterators \pause[]
        \item \texttt{vector<int>::iterator it = v.begin();} returns a \textit{Pointer} to the first element of vector, \pause[]
        \item Or you can use \texttt{auto it = v.begin();}\pause[]
        \item For example if you have an array $v = [10, 12, 13,14]$ then \texttt{*it} would return $10$,\pause[]
        \item Similar to pointer you can increase and decrease them, \texttt{it++;} and then \texttt{*it} would return $12$.
    \end{itemize}

\end{frame}

\begin{frame}
    \frametitle{Last Iterator}

    
    \begin{itemize}
        \item \texttt{vector<int>::iterator it = v.end();} points to a non-existent element sits after the last element.\pause[]
        \item Hence \texttt{*it} would dereference nothing when \texttt{it = v.end();}.
    \end{itemize}

\end{frame}

\begin{frame}
    \frametitle{\texttt{std::lower\_bound} and \texttt{std::upper\_bound}}

    
    \begin{itemize}
        \item We need a non-decreasingly sorted container,
        \item We want to find out position of the smallest number just $>$ (greater) a given number or position of the smallest number $\geq$ (greater than or equal to) a given number
    \end{itemize}

\end{frame}

\begin{frame}
    \frametitle{\texttt{std::lower\_bound} and \texttt{std::upper\_bound}}

    
    \begin{itemize}
        \item \texttt{std::lower\_bound} returns iterator to first element in the given range which is equal or greater than the value.
        \item \texttt{std::upper\_bound} returns iterator to first element in the given range which is greater than the value.
    \end{itemize}

\end{frame}
